#include <stdio.h>

int main(void)
포인터 사용----------------------------------------------------
{
	int x = 10;
	int* p = &x;

	printf("x= %d\n", x); //10
	printf("&x=%p\n", &x); //x주소

	printf("p=%p\n", p); //x주소
	printf("*p=%d\n", *p); //10
	printf("&p=%p\n", &p);//p주소

	*p = 20;
	printf("*p=%d\n", *p); //20

	return 0;
}

포인터 증감------------------------------------------------------
int main(void)
{
	int arr[5] = { 1,2,3,4,5 };
	int* p = &arr[0];
	//int* p = arr; (위랑 같음)

	for (int i = 0; i < 5; i++, p++)
	{
		printf("p=%p, ", p);
		printf("p=%d\n", *p);
	}
	p = &arr[0]; //이거 안써도 됨

	for (int i = 0; i < 5; i++) //이거 먼저쓰면 
	{
		printf("p=%p, ", p + i);
		printf("p=%d\n", *(p + i));
	}
}

int main(void)
{
	int arr[5] = { 1,2,3,4,5 };
	int* p = arr;
	int i;

	for (i = 0; i < 5; i++) {
		printf("%d ", *(arr + i)); //배열 이름을 포인터인 것 처럼 값 출력
		printf("%p\n ", arr + i); //주소를 출력
	}
	return 0;
}
